## JavaScript — app.js

Create `web/static/js/app.js` with minimal functionality:

- Connect chat input to POST `/ask` (or WS) and append messages to the scrollable history.
- Wire "CONTACT" click to POST `/contact` with the latest N messages and current user email (if any).
- Keep chat history scrolling within the container; input stays at bottom.

Example:

```javascript
// web/static/js/app.js
async function sendChat() {
    const input = document.getElementById('chat-input');
    const text = input.value.trim();
    if (!text) return;
    appendMessage('user', text);
    input.value = '';

    const res = await fetch('/ask', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ query: text })
    });
    const data = await res.json();
    appendMessage('agent', data.answer || '(no answer)');
}

function appendMessage(role, text) {
    const stream = document.getElementById('chat-stream');
    const div = document.createElement('div');
    div.className = 'msg ' + role;
    div.textContent = text;
    stream.appendChild(div);
    stream.scrollTop = stream.scrollHeight;
}

async function contact() {
    const payload = { reason: 'user_clicked_contact' };
    await fetch('/contact', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
    });
    alert('We have emailed support.');
}

document.getElementById('chat-send').addEventListener('click', sendChat);
const contactBtn = document.getElementById('contact-btn');
if (contactBtn) contactBtn.addEventListener('click', contact);
```
